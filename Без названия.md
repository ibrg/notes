@app.route('/posts')

def posts_list():
	page = request.args.get('page', default=1, type=int)
	per = 10
	bottom = (page - 1) * per
	top = bottom + per
	posts = repo.posts[bottom:top]

	return render_template(
		'posts/index.html',
		posts=posts,
		page=page)


@app.get('/post/<slug>')
def post_detail(slug):
	post = repo.posts.find(slug)
	return render_template('posts/show.html', post)
